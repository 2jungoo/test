# 컴파일러와 옵션 지정
CC = gcc
CFLAGS = -g -Wall

# 최종 실행 파일 이름
TARGET = file_analyzer

# 소스 파일 목록
SRCS = main.c print.c word.c

# 소스 파일로부터 생성될 오브젝트 파일 목록
OBJS = $(SRCS:.c=.o)

# 최종 타겟 빌드 규칙
$(TARGET): $(OBJS)
	$(CC) $(CFLAGS) -o $(TARGET) $(OBJS)

# .c 파일로부터 .o 파일을 만드는 규칙
%.o: %.c file_utils.h
	$(CC) $(CFLAGS) -c $< -o $@

# 이전 빌드 결과물 삭제
clean:
	rm -f $(TARGET) $(OBJS)

.PHONY: clean
