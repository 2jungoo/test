#include <stdio.h>
#include <stdlib.h>
#include "file_utils.h"

void printfile(const char *filename) {
    FILE *fp;
    char buffer[1024];

    if ((fp = fopen(filename, "r")) == NULL) {
        perror(filename);
        return;
    }

    printf("--- 파일 내용 ---\n");
    while (fgets(buffer, sizeof(buffer), fp) != NULL) {
        printf("%s", buffer);
    }
    printf("-----------------\n\n");

    fclose(fp);
}
