#include <stdio.h>

void swap_value(int a, int b) {
    int temp = a;
    a = b;
    b = temp;
    // 이 함수 안에서만 a와 b의 값이 바뀜
}

int main() {
    int num1 = 100;
    int num2 = 200;
    printf("교환 전: num1 = %d, num2 = %d\n", num1, num2);
    swap_value(num1, num2); 
    printf("교환 후: num1 = %d, num2 = %d\n", num1, num2);
    return 0;
}
실행결과(swap 실패)
교환 전: num1 = 10, num2 = 20
교환 후: num1 = 10, num2 = 20
