#include <stdio.h>
void swap(int *a, int *b) {
    int temp = *a; // a가 가리키는 곳의 값(원본)을 temp에 저장
    *a = *b;       // b가 가리키는 곳의 값을 a가 가리키는 곳에 저장
    *b = temp;     // temp의 값을 b가 가리키는 곳에 저장
}
int main() {
    int num1 = 100;
    int num2 = 200;
    printf("교환 전: num1 = %d, num2 = %d\n", num1, num2);
    swap(&num1, &num2); 
    printf("교환 후: num1 = %d, num2 = %d\n", num1, num2);
    return 0;
}
